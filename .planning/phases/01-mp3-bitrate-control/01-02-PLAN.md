```typescript
export function synthSubmit(reqs: AceRequest[], format: string, mp3Bitrate = 0): Promise<string> {
    const url = format !== 'mp3' ? `synth?format=${format}` : 'synth';
    // Inject mp3_bitrate into the request object(s) when encoding as MP3 with a valid preset bitrate.
    // mp3_bitrate is a ServerFields key, not an AceRequest field — inject at serialization time only.
    const inject = (format === 'mp3' && mp3Bitrate > 0) ? { mp3_bitrate: mp3Bitrate } : {};
    const single = { ...reqs[0], ...inject };
    const body = reqs.length === 1 ? JSON.stringify(single) : JSON.stringify(reqs.map(r => ({ ...r, ...inject })));
    return submitJob(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body
    });
}
```

**Updated synthSubmitWithAudio:**
```typescript
export function synthSubmitWithAudio(
    reqs: AceRequest[],
    srcAudio: Blob | null,
    refAudio: Blob | null,
    format: string,
    mp3Bitrate = 0
): Promise<string> {
    const url = format !== 'mp3' ? `synth?format=${format}` : 'synth';
    const inject = (format === 'mp3' && mp3Bitrate > 0) ? { mp3_bitrate: mp3Bitrate } : {};
    const single = { ...reqs[0], ...inject };
    const body = reqs.length === 1 ? JSON.stringify(single) : JSON.stringify(reqs.map(r => ({ ...r, ...inject })));
    const form = new FormData();
    form.append('request', new Blob([body], { type: 'application/json' }), 'request.json');
    if (srcAudio) form.append('audio', srcAudio, 'src.audio');
    if (refAudio) form.append('ref_audio', refAudio, 'ref.audio');
    return submitJob(url, { method: 'POST', body: form });
}
```

Both functions now accept an optional `mp3Bitrate: number = 0` parameter. When format is 'mp3' and mp3Bitrate > 0, `mp3_bitrate` is spread into the serialized request object(s). When format is not 'mp3' or mp3Bitrate is 0, the inject object is empty — no mp3_bitrate key in the body. The default value of 0 preserves backward compatibility for any existing callers that do not pass the argument (the server treats 0 as "use server default" per SRV-03).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| localStorage → load() | Persisted mp3Bitrate could be any JSON value from a tampered storage entry |
| app.mp3Bitrate → JSON body | Client-side value injected into request sent to server |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | load(): parsed.mp3Bitrate from localStorage | mitigate | Validate against curated preset list [96,128,160,192,256,320]; default to 0 if not in list. 0 triggers server-default fallback. |
| T-02-02 | Tampering | synthSubmit/WithAudio: mp3_bitrate in JSON body | accept | Server validates the value independently against its own allowlist (Plan 01). Client-side restriction to presets is UX, not a security boundary. |
| T-02-03 | Spoofing | AceRequest type: mp3_bitrate added to type | mitigate | mp3_bitrate is deliberately excluded from AceRequest. It is injected at serialization time only. This prevents mp3_bitrate from leaking into IndexedDB song records, batch arrays, or LM workers where it has no meaning. |
